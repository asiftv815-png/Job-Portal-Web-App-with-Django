#include<iostream>
using namespace std;

class Node{
      public:
      int data;
      Node* next;
      Node* prev;

      Node(int val){
          data = val;
          next = nullptr;
          prev = nullptr;
      }
};

class DLL{
      private:
      Node* head;

      public:
      DLL(){
          head = nullptr;
      }

      void insertF(int val){
            Node* newNode = new Node(val);
            if(head == nullptr){
                head = newNode;
                return;
            }
            newNode->next = head;
            head->prev = newNode;
            head = newNode;
      }

      void insertL(int val){
            Node* newNode = new Node(val);
            if(head == nullptr){
                  head = newNode;
                  return;
            }
            Node* temp = head;
            while(temp->next != nullptr){
                  temp = temp->next;
            }
            temp->next = newNode;
            newNode->prev = temp;
      }
      void insert(int val, int pos){
            Node* newNode = new Node(val);
            if(pos <= 0){
                  cout << "mara khao"<< endl;
                  return;
            }
            if(pos == 1){
                  insertF(val);
                  return;
            }
            Node* temp = head;
            for(int i = 1; i < pos-1; i++){
                  if(temp == nullptr){
                        cout << "Position out of bounds"<< endl;
                        return;
                  }
                  temp = temp->next;
            }
            if(temp == nullptr){
                  cout << "Position out of bounds"<< endl;
                  return;
            }
            newNode->next = temp->next;
            newNode->prev = temp;
            if(temp->next != nullptr){
                  temp->next->prev = newNode;
            }
            temp->next = newNode;

      }

      void delf(){
            if(head == nullptr){
                  cout << "List is empty"<< endl;
                  return;
            }
            if(head->next == nullptr){
                  delete head;
                  head = nullptr;
                  return;
            }
            
            Node* temp = head;
            head = head->next;
            head->prev = nullptr;
            delete temp;
            
      }

      void delb(){
            if(head == nullptr){
                  cout << "List is empty"<< endl;
                  return;
            }
            if(head->next == nullptr){
                  delf();
                  return;
            }
            Node* temp = head;
            while(temp->next != nullptr){
                  temp = temp->next;
            }
            temp = temp->prev;
            delete temp->next;
            temp->next = nullptr;
      }

      void display(){
          Node* temp = head;
          while(temp != nullptr){
              cout << temp->data << "->";
              temp = temp->next;
          }
          cout << "Null"<< endl;
      }
      
      void dell(int pos){
            if(head == nullptr){
                  cout << "List is empty"<< endl;
                  return;
            }
            if(pos <= 0){
                  cout << "Invalid position"<< endl;
                  return;
            }
            if(pos == 1){
                  delf();
                  return;
            }
            Node* temp = head;
            for(int i = 1 ; i<pos-1; i++){
                  if(temp == nullptr || temp->next == nullptr){
                        cout << "Position out of bounds"<< endl;
                        return;
                  }
                  temp = temp->next;
            }

            Node* del = temp->next;
            if(del == nullptr){
                  cout << "Position out of bounds"<< endl;
                  return;
            }
            temp->next = del->next;
            if(del->next != nullptr){
                  del->next->prev = temp;
            }
            delete del;

      }
};

int main(){
      DLL lt;
      lt.insertF(45);
      lt.insertF(98);
      lt.insertF(12);
      lt.display();
      lt.insertL(77);
      lt.insertL(33);
      lt.display();
      lt.insert(100, 3);
      lt.display();
      lt.dell(3);
      lt.display();

      return 0;
}
